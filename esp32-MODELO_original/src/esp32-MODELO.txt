#include "Sensores.hpp"

void setup() {
  Serial.begin(115200);

  LED::init();
  Sensor::init();
}
void loop() {
  float temp = Sensor::readTemp();
  Serial.printf("Temperatura: %.2f °C\n", temp);
  // Exemplo de alias para funções do LED
  namespace L = LED;
  (temp > 30) ? L::on() : L::off();
  delay(2000);
}
